图像阈值

 res , dst =cv2.threshold(scr,thresh,maxval,type)


 #scr:只能输入单通道图像 一般为灰度图 thresh：阈值 maxval：最大值 type 二值操作类型
 #res：阈值  thresh：输出图

#亮的->最亮 暗的->最暗
ret , thresh1 = cv2.threshold(img, 127 , 255 , cv2.THRESH_BINARY)
#亮的->最暗 暗的->最亮
ret , thresh2 = cv2.threshold(img, 127 , 255 , cv2.THRESH_BINARY_INV)
#亮的->中等 暗的->不变
ret , thresh3 = cv2.threshold(img, 127 , 255 , cv2.THRESH_TRUNC)
#亮的->不变 暗的->最暗
ret , thresh4 = cv2.threshold(img, 127 , 255 , cv2.THRESH_TOZERO)
#亮的->最亮 暗的->不变
ret , thresh5 = cv2.threshold(img, 127 , 255 , cv2.THRESH_TOZERO_INV)

#打印
titles = ['Original Image','BINARY','BINARY_INV', 'TRUNC', 'TOZERO', 'TOZERO_INV']

腐蚀

#变细变小

# 变大

开运算与闭运算

# 开运算： 先腐蚀，后膨胀  MORPH_OPEN
kernel = np.ones((5,5),np.uint8)
cv2.morphologyEx(img ,cv2.MORPH_OPEN,kernel)

# 闭运算： 先膨胀，后腐蚀  MORPH_CLOSE
kernel = np.ones((5,5),np.uint8)
cv2.morphologyEx(img ,cv2.MORPH_CLOSE,kernel)

梯度运算

# 膨胀 - 腐蚀   MORPH_GRADIENT    轮廓，边界
kernel = np.ones((5,5),np.uint8)
cv2.morphologyEx(img,cv2.MORPH_GRADIENT,kernel)

礼帽与黑帽

#礼貌： 原始输入 - 开运算结果    只剩刺
kernel = np.ones((5,5),np.uint8)
cv2.morphologyEx(img,cv2.MORPH_TOPHAT,kernel)

# 黑帽 ： 闭运算 - 原始输入     原始轮廓
kernel = np.ones((5,5),np.uint8)
cv2.morphologyEx(img,cv2.MORPH_BLACKHAT,kernel)

梯度计算

sobel算子

       | P1  P2  P3 |
  A  = | P4  P5  P6 |    
       | P7  P8  P9 |
# x轴计算   右 - 左
       | -1  0  1 |
  Gx = | -2  0  2 |           P5x =  （P3 - P1） + 2*（P6 - P4） + （P9 - P7）
       | -1  0  1 |
# y轴计算   下 - 上
       | -1  -2  -1 |
  Gy = |  0   0   0 |         P5y =  （P7 - P1） + 2*（P8 - P2） + （P9 - P3）
       |  1   2   1 |
# G = |Gx| + |Gy|

Code:

# 水平方向
cv2.Sobel(img,cv2.CV_64F,1,0,ksize=3) 
# cv2.CV_64F :位数变多，能表示负数  水平方向（1：水平  0：竖直 ）ksize：大小-3x3
#白到黑是正数，黑到白是负数，所有负数会截断成0，要取绝对值
cv2.convertScaleAbs(img) #取绝对值

# 竖直方向
 cv2.Sobel(img,cv2.CV_64F,0,1,ksize=3) 
#cv2.CV_64F :能表示负数  竖直方向（1：水平  0：竖直 ）ksize：大小-3x3
#白到黑是正数，黑到白是负数，所有负数会截断成0，要取绝对值
cv2.convertScaleAbs(img) #取绝对值

#图像结合
cv2.addWeighted(img1,0.5,img2,0.5,0)

Scharr算子

       | P1  P2  P3 |
  A  = | P4  P5  P6 |    
       | P7  P8  P9 |
# x轴计算   右 - 左
       |  -3  0   3 |
  Gx = | -10  0  10 |           P5x =  3*（P3 - P1） + 10*（P6 - P4） + 3*（P9 - P7）
       |  -3  0   3 |
# y轴计算   下 - 上
       | -3  -10  -3 |
  Gy = |  0    0   0 |         P5y =   3*（P7 - P1） + 10*（P8 - P2） + 3*（P9 - P3）
       |  3   10   3 |
# G = |Gx| + |Gy|

Code:

#水平方向
cv2.Scharr(img,cv2.CV_64F,1,0)
cv2.convertScaleAbs(img)

#竖直方向
cv2.Scharr(img,cv2.CV_64F,0,1)
cv2.convertScaleAbs(img)

#图像结合
cv2.addWeighted(img1,0.5,img2,0.5,0)

laplacian算子

       | P1  P2  P3 |
  A  = | P4  P5  P6 |    
       | P7  P8  P9 |

       |   0   0   0 |
  G  = |   1  -4   1 |         
       |   0   1   0 |

G = P2 + P4 + P6 + P8 - 4P5

Code:

cv2.Laplacian(img,cv2.CV_64F) #没有x y 不需要合并
cv2.convertScaleAbs(img)
