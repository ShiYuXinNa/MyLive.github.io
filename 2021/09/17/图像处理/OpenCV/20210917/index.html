#图像平滑处理

##均值滤波
#简单的平均卷积操作
cv2.blur(img,(3,3)) # 取3×3像素

##方框滤波
#基本和均值滤波一样 ， 可以归一化
#-1 一致 True :归一化 和均值滤波一样 False :容易越界
cv2.boxFilter(img,-1,(7,7),normalize=True)

##高斯滤波
#高斯模糊的卷积里的数值是满足高斯分布，相当于更重视中间的 （越近权重越高）
cv2.GaussianBlur(img,(3,3),1)

##中值滤波
#用中值代替 (5x5)
cv2.medianBlur(img,5)

#形态学操作
##腐蚀操作 变细变小
kernel = np.ones((3,3),np.uint8) //3x3
cv2.erode(img,kernel,iterations = 1) #kernel:盒 iteration:迭代次数
##膨胀操作 变大
kernel = np.ones((3,3),np.uint8)
cv2.dilate(img,kernel,iterations=1)

#有细条的文字可通过腐蚀 再膨胀处理

##开运算 ：先腐蚀，后膨胀 MORPH_OPEN
kernel = np.ones((5,5),np.uint8)
cv2.morphologyEx(img ,cv2.MORPH_OPEN,kernel)

##闭运算 ：先膨胀，后腐蚀 MORPH_CLOSE
kernel = np.ones((5,5),np.uint8)
cv2.morphologyEx(img ,cv2.MORPH_CLOSE,kernel)

##梯度运算 :膨胀 - 腐蚀 MORPH_GRADIENT 轮廓，边界
kernel = np.ones((5,5),np.uint8)
cv2.morphologyEx(img,cv2.MORPH_GRADIENT,kernel)

##礼帽 ： 原始输入 - 开运算结果 只剩刺
kernel = np.ones((5,5),np.uint8)
cv2.morphologyEx(img,cv2.MORPH_TOPHAT,kernel)

##黑帽 ： 闭运算 - 原始输入 原始轮廓。。
kernel = np.ones((5,5),np.uint8)
cv2.morphologyEx(img,cv2.MORPH_BLACKHAT,kernel)

#梯度计算

sobel算子

       | P1  P2  P3 |
  A  = | P4  P5  P6 |    
       | P7  P8  P9 |
# x轴计算   右 - 左
       | -1  0  1 |
  Gx = | -2  0  2 |           P5x =  （P3 - P1） + 2*（P6 - P4） + （P9 - P7）
       | -1  0  1 |
# y轴计算   下 - 上
       | -1  -2  -1 |
  Gy = |  0   0   0 |         P5y =  （P7 - P1） + 2*（P8 - P2） + （P9 - P3）
       |  1   2   1 |
# G = |Gx| + |Gy|

###Code:

#水平方向
cv2.Sobel(img,cv2.CV_64F,1,0,ksize=3) 
# cv2.CV_64F :位数变多，能表示负数  水平方向（1：水平  0：竖直 ）ksize：大小-3x3
#白到黑是正数，黑到白是负数，所有负数会截断成0，要取绝对值
cv2.convertScaleAbs(img) #取绝对值

#竖直方向
 cv2.Sobel(img,cv2.CV_64F,0,1,ksize=3)

#白到黑是正数，黑到白是负数，所有负数会截断成0，要取绝对值
cv2.convertScaleAbs(img) #取绝对值

Scharr算子

       | P1  P2  P3 |
  A  = | P4  P5  P6 |    
       | P7  P8  P9 |
# x轴计算   右 - 左
       |  -3  0   3 |
  Gx = | -10  0  10 |           P5x =  3*（P3 - P1） + 10*（P6 - P4） + 3*（P9 - P7）
       |  -3  0   3 |
# y轴计算   下 - 上
       | -3  -10  -3 |
  Gy = |  0    0   0 |         P5y =   3*（P7 - P1） + 10*（P8 - P2） + 3*（P9 - P3）
       |  3   10   3 |
# G = |Gx| + |Gy|

###Code:
#水平方向
cv2.Scharr(img,cv2.CV_64F,1,0)
cv2.convertScaleAbs(img)

#竖直方向
cv2.Scharr(img,cv2.CV_64F,0,1)
cv2.convertScaleAbs(img)

#图像结合
cv2.addWeighted(img1,0.5,img2,0.5,0)

laplacian算子

       | P1  P2  P3 |
  A  = | P4  P5  P6 |    
       | P7  P8  P9 |

       |   0   0   0 |
  G  = |   1  -4   1 |         
       |   0   1   0 |

G = P2 + P4 + P6 + P8 - 4P5

####Code:
cv2.Laplacian(img,cv2.CV_64F) #没有x y 不需要合并
cv2.convertScaleAbs(img)

#Canny边缘检测

  1. 使用高斯滤波器，以平滑图像，滤除噪声。
  2. 计算图像中每个像素点的梯度强度和方向。
  3. 应用非极大值抑制，以消除边缘检测带来的杂散响应。

    1）线性插值法 ：M（q） = w * M（g2） + （1-w）* M（g1）
     w = distance（q,g2) / distance (g1,g2) </br>
    1. 为了简化计算，由于一个像素周围有8个像素，把一个像素的梯度方向离散为8个方向，这样只需要计算前后，不用插值
  4. 应用双阈值检测来确定真实和潜在的边缘。
  5. 通过抑制孤立的弱边缘最终完成边缘检测。
